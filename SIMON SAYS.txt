correctButtonChecker(){
    if (currentPrompt = pulling && reelIn.isDown){
        standardDamage();
    } else if (currentPrompt = pulling && reelIn.isDown){
        maxDamage();
    } else if (currentPrompt = movingLeft && standLeft || currentPrompt = movingRight && standRight) {
        fishStun();
    }
    } else {
        lineTension += 5;
    }


}

standardDamage(){
    fishHP - 5;
    lineTension += 5
    if (fishHP <= 0) {
        catchFish();
    }
}

maxDamage(){
    fishHP - 70;
    lineTension += 10
    if (fishHP <= 0) {
        catchFish();
    }
}
